    current_table->big_blind_position

Mutexes (must use)

    // When accessing game state:
    pthread_mutex_lock(&current_table->state_mutex);

    // ... access state ...
    pthread_mutex_unlock(&current_table->state_mutex);

    // When accessing players:
    pthread_mutex_lock(&current_table->players_mutex);
    // ... access players ...
    pthread_mutex_unlock(&current_table->players_mutex);

    // When accessing pot/bets/community cards:
    pthread_mutex_lock(&current_table->game_mutex);
    // ... access game data ...
    pthread_mutex_unlock(&current_table->game_mutex);

    // When accessing turn:
    pthread_mutex_lock(&current_table->turn_mutex);
    // ... access turn ...
    pthread_mutex_unlock(&current_table->turn_mutex);

Condition Variables that you can Signal
